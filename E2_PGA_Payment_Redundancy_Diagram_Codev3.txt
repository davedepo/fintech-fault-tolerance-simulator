    <script>
        let failureActive = false;
        
        function startTransaction() {
            // Animate the regions
            const regions = document.querySelectorAll('.region-circle');
            regions.forEach((region, index) => {
                setTimeout(() => {
                    region.classList.add('pulsing');
                    setTimeout(() => {
                        region.classList.remove('pulsing');
                    }, 2000);
                }, index * 200);
            });
            
            // Update response time
            document.getElementById('response-time').textContent = '< 450ms';
            setTimeout(() => {
                document.getElementById('response-time').textContent = '< 500ms';
            }, 3000);
        }
        
        function simulateFailure() {
            if (failureActive) return;
            
            failureActive = true;
            
            // Show failure overlay
            document.getElementById('failure-overlay').classList.add('active');
            
            // Mark US East region and data center as failed (RED)
            const usEast = document.getElementById('region-east');
            const dcEast = document.getElementById('dc-east');
            const dbPrimary = document.getElementById('db-primary');
            
            usEast.classList.add('failed');
            dcEast.classList.add('failed');
            dbPrimary.classList.add('failed');
            
            // Update health status for failed components
            dcEast.querySelector('.health-monitor').textContent = '❌ System Down';
            
            // Update servers count
            document.getElementById('active-servers').textContent = '8/13';
            
            setTimeout(() => {
                document.getElementById('failure-overlay').classList.remove('active');
                
                // Activate US West as backup (GREEN)
                const usWest = document.getElementById('region-west');
                const dcWest = document.getElementById('dc-west');
                const dbStandby = document.getElementById('db-standby');
                
                usWest.classList.add('backup-active');
                dcWest.classList.add('backup-active');
                dbStandby.classList.add('backup-active');
                
                // Update labels for activated backup
                dbStandby.querySelector('.db-icon').classList.remove('standby');
                dbStandby.querySelector('.db-icon').classList.add('primary');
                dbStandby.querySelector('.db-label').textContent = 'Primary DB';
                dbStandby.querySelector('.db-info').innerHTML = 'US West<br>Read/Write';
                
                dcWest.querySelector('.health-monitor').textContent = '✅ Now Primary';
                usWest.querySelector('.region-info').innerHTML = 'California<br>PRIMARY ACTIVE';
                
                // Update response time during failover
                document.getElementById('response-time').textContent = '< 520ms';
                setTimeout(() => {
                    document.getElementById('response-time').textContent = '< 500ms';
                }, 2000);
                
            }, 2000);
        }
        
        function resetSystem() {
            failureActive = false;
            
            // Reset all visual elements
            document.getElementById('failure-overlay').classList.remove('active');
            
            const usEast = document.getElementById('region-east');
            const dcEast = document.getElementById('dc-east');
            const usWest = document.getElementById('region-west');
            const dcWest = document.getElementById('dc-west');
            const dbStandby = document.getElementById('db-standby');
            const dbPrimary = document.getElementById('db-primary');
            
            // Remove all failure and backup classes
            usEast.classList.remove('failed');
            dcEast.classList.remove('failed');
            dbPrimary.classList.remove('failed');
            usWest.classList.remove('backup-active');
            dcWest.classList.remove('backup-active');
            dbStandby.classList.remove('backup-active');
            
            // Reset database states
            dbStandby.querySelector('.db-icon').classList.remove('primary');
            dbStandby.querySelector('.db-icon').classList.add('standby');
            dbStandby.querySelector('.db-label').textContent = 'Hot Standby';
            dbStandby.querySelector('.db-info').innerHTML = 'US West<br>Ready to takeover';
            
            // Reset health monitors and labels
            dcEast.querySelector('.health-monitor').textContent = '💙 All Healthy';
            dcWest.querySelector('.health-monitor').textContent = '💙 All Healthy';
            usWest.querySelector('.region-info').innerHTML = 'California<br>Secondary Route';
            
            // Reset metrics
            document.getElementById('active-servers').textContent = '13/13';
            document.getElementById('response-time').textContent = '< 500ms';
        }
        
        // Add hover effects for educational purposes
        document.querySelectorAll('.region, .server, .db-icon').forEach(element => {
            element.addEventListener('mouseenter', function() {
                this.style.transform = 'scale(1.05)';
            });
            
            element.addEventListener('mouseleave', function() {
                this.style.transform = 'scale(1)';
            });
        });
    </script>
